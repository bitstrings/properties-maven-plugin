properties-maven-plugin
=======================

```xml
<groupid></groupId>
<artifactId></artifactId>
<executions>
    <execution>
        <goals>
            <goal>properties</goal>
        </goals>
        <configuration>
            <define>
                <dependencies>
                    <select>
                        <includes/>
                        <excludes/>
                        <transitive/>
                        <scope/>
                    </select>
                    <format>
                        <namePattern/>
                        <valuePattern/>
                    </format>
                </dependencies>
            </define>
            <set>
                <select>
                    <includes/>
                    <excludes/>
                </select>
                <target/>
                <override/>
            </set>
            <write>
                <select>
                    <includes/>
                    <excludes/>
                </select>
                <files>
                    <file>
                        <path/>
                        <append/>
                    </file>
                </files>
            </write>
        </configuration>
    <execution>
<executions>
```
